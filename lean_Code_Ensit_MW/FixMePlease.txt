function isPassing(student, minGrade) {
  return student.score >= minGrade;
}


function handleSuspension(student) {
  if (student.failedBefore) {
        console.log(
          student.gender === Gender.Male
            ? `Sorry Mr. ${student.name}, you are suspended.` 
            : `Sorry Mrs. ${student.name}, you are suspended.` 
        );
  }else {
          student.failedBefore = true; 
      }
}

function evaluateStudents(studentsList) {
  const MINIMUM_GRADE_TO_PASS = 10;
  let listPassedStudents = [];

  studentsList.forEach((student) => {
    if (isPassing(student, MINIMUM_GRADE_TO_PASS)) {
      listPassedStudents.push(student);
    } else {
      handleSuspension(student);
    }
  });

  return listPassedStudents;
}
