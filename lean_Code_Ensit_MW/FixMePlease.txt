        break;
    case Gender.Female:

//Q10: Who wants to have a function that do multiple things with multiple comments?
function checkIfStudentPassed(student, minimumGradeToPass) {
  return student.score >= minimumGradeToPass;
}
function studentFailedMessage(student) {
  if (student.failedBefore) {
    console.log(
      student.gender === Gender.Male
        ? `Sorry Mr. ${student.name}, you are suspended.` 
        : `Sorry Mrs. ${student.name}, you are suspended.` 
    );
  } else {
    student.failedBefore = true;
  }
}
function evaluateStudents(studentsList) {
  let passedStudents = []; 
  const minimumGradeToPass = 10; 
  studentsList.forEach((student) => {
    if (checkIfStudentPassed(student , minimumGradeToPass)) {
      passedStudents.push(student); // Here we add students
    } else {
      if (student.failedBefore) {
        studentFailedMessage(student) ;
      }
    }
  });

  return passedStudents;
}
