function isPassed(student, passedStudents){
  return student.score >= MINIMUM_GRADE_TO_PASS;
}

function setFailedBefore(student) {
  student.failedBefore = true;
}

function evaluateStudents(studentsList) {
  let passedStudents = [];
  const MINIMUM_GRADE_TO_PASS = 10;
  
  studentsList.forEach((student) => {
    if ( isPassed(student) ) {
      passedStudents.push(student);
    } else {
      if ( !student.failedBefore ) {
        setFailedBefore(student)
        displaySuspensionMessage(student);
      }
    }
  });

  return passedStudents;
}
